create table golfapp.COMPETITION(
    id int auto_increment,
    name varchar(40) not null, 
    primary key(id)
);

create table golfapp.PLAYER(
    id int auto_increment,
    name varchar(40) not null, 
    username varchar(40) not null,
    handicap int,
    primary key(id)
);

create table golfapp.GOLF_LEADER_BOARD(
    id int auto_increment,
    competition_id int ,
    player_id int,
    avg_score_per_round int,
    best_five_total_round int,
    bonus_rounds int,
    total_points int,
    total_score int,
    primary key(id),
    foreign key(competition_id)
        references competition(id)
);

create table golfapp.GOLF_EVENT(
    id int auto_increment,
    competition_id int ,
    name varchar(40) not null,
    venue varchar(40),
    type varchar(20),
    date date,
    primary key(id),
    foreign key(competition_id)
          references competition(id)
);

create table golfapp.SCORE(
    id int auto_increment,

    player_id int,
    event_id int,
    par int,
    score int,
    points int,

    primary key(id),

          references competition(id)
